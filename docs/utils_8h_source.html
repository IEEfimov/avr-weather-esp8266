33 #define _bv(num) (1 << num) // Integer to binary shift
34 #define sbi(reg, bit) ( reg |= _bv(bit) ) // Set bit
35 #define cbi(reg, bit) ( reg &= ~_bv(bit) ) // Clear bit
36 #define tbi(reg, bit) ( reg ^= _bv(bit) ) // Toggle bit
37 #define wbi(reg, bit, value) ( value ? sbi(reg, bit) : cbi(reg, bit) ) // Write bit
